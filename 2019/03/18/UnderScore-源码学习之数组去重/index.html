UnderScore 源码学习之数组去重

Posted at 2019-03-18 UnderScore 

数组去重的方法非常多，这里列举其中一些以及 underscore 的方法

方法一

最符合直觉的将原数组元素和结果数组一一比较

function unique(a){
  var res = []
  for(i=0;i<a.length;i++){
    var item = a[i]
    for(var j=0,jLen=res.length;j<jLen;j++){
      if(res[j] === item)
      break
    }
    if(j===jLen) res.push(item)
  }
  return res
}

indexOf

function unique(a){
  var res = []
  for(i=0;i<a.length;i++){
    var item = a[i]
    // indexOf 值为 -1 表示未找到
    (res.indexOf(item)=== -1)&& res.push(item)
  }
  return res
}

filter

function unique(a){
    // filter 回调函数接收三个参数
  var res = a.filter(function(item,index,array){
    return array.indexOf(item) === index
  })
  return res
}

复杂度 O(n^2)

方法二

依次将原数组重复元素的最后一个元素放入结果数组

function unique(a){
  var res = []
  for(var i = 0,len = a.length; i < len;i++){
    for(var j = i + 1; j<len ;j j++){
      // 发现相同元素，则 i 自增进入下个内循环比较 a[i+1] 和后面的元素，
      // 直到找到重复元素的最后一个，即后面再没有和它重复的元素，或非重复元素的第一个
      if(a[i] === a[j])
        j = ++i
    }
    res.push(a[i])
  }
  return res
}

复杂度 O(n^2)

方法三：sort

sort 排序后，理论上相同的元素位置上相邻，可以通过比较前后位置的元素去重

function unique(a){
  a.concat().sort().filter(function(item, pos, arr){
    return !pos || item != arr[pos-1]
  })
}

但是如果不同元素的 toString() 结果相同会导致错误

方法四：object

用 Object 对象来当作哈希表

function unique(a){
  var seen = {}
  return a.filter(function(item){
    return seen.hasOwnProperty(item) ? false : (seen[item] = true)
  })
}

方法五：ECMAScript 6 新增

function unique(a){
  return Array.from(new Set(a))
}

需要浏览器支持
